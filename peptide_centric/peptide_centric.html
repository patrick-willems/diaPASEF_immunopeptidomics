Peptide-centric analysis with DIA-NN 2.0.2

DIA-NN 2.0.2 predicted libraries with all 8-12 mers were constructed (allowing one methionine oxidation) per charge state (1-3+). These were searched independently in proteoform scoring mode, using predicted spectra for precursors present in the FragPipe spectrum-centric analysis library.tsv for calibration. The search results (report.parquet) were then used as empirical DIA library in a second pass search in generic scoring mode - the main report of that is used here.

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from Bio import SeqIO
from tqdm import tqdm
import sys

# We will re-map the stripped.sequences.
fasta_path = 'EGD_human.fasta'
records = list(SeqIO.parse(fasta_path, "fasta"))
fasta_dict = {record.id: str(record.seq) for record in records}

# Read DIA-NN main report and get unique sequences.
df = pd.read_parquet('report.parquet')
pepcentric_peps = set(df["Stripped.Sequence"].unique())

# Map the sequences to proteins
mapping_dict = {}
for pep in tqdm(pepcentric_peps, desc="Mapping peptides to proteins"):
    matching_proteins = [prot_id for prot_id, seq in fasta_dict.items() if pep in seq] 
    mapping_dict[pep] = ";".join(matching_proteins) if matching_proteins else "No_match"
df['Mapped_Proteins'] = df['Stripped.Sequence'].apply(lambda pep: mapping_dict.get(pep, 'No_match'))

# Log2 transform.
df["log2_int"] = np.log2(df["Precursor.Normalised"]) 

# Peptide found in DDA or not
df['group'] = 'pep_centric'
df.loc[df['Stripped.Sequence'].isin(spectrum_centric_peps), 'group'] = 'spec_centric'
df.loc[df['Stripped.Sequence'].isin(dda_peps), 'group'] = 'dda'

# Pivot intensities.
pivot_df = df.pivot_table(
    index=['Precursor.Id', 'Protein.Ids','Protein.Names','Mapped_Proteins','Genes','Stripped.Sequence','group'],
    columns='Run',
    values='log2_int'
)

# Median normalize per sample (column-wise)
pivot_df = pivot_df.subtract(pivot_df.median(axis=0), axis=1)

# Save matrix.
pivot_df.to_csv('Pepcentric_DIANN_matrix.tsv',sep='\t') # Save values.

# Density plot of DIA-only peptides and those also found by ddaPASEF.
dda = pivot_df[pivot_df.index.get_level_values('group') == 'dda'].values.flatten()
spec_centric = pivot_df[pivot_df.index.get_level_values('group') == 'spec_centric'].values.flatten()
pep_centric = pivot_df[pivot_df.index.get_level_values('group') == 'pep_centric'].values.flatten()
dda = dda[~np.isnan(dda)]
spec_centric = spec_centric[~np.isnan(spec_centric)]
pep_centric = pep_centric[~np.isnan(pep_centric)]

plt.figure(figsize=(8, 6))
sns.kdeplot(dda, fill=True, label="DDA", color="dodgerblue", linewidth=2, alpha=0.2)
sns.kdeplot(spec_centric, fill=True, label="Spectrum-centric", color="forestgreen", linewidth=2, alpha=0.2)
sns.kdeplot(pep_centric, fill=True, label="Peptide-centric", color="firebrick", linewidth=2, alpha=0.2)
plt.xlabel("log2 Normalized Intensity")
plt.ylabel("Density")
plt.title("Pep-centric search DIANN: Peptide Intensity Distribution")
plt.legend()
plt.tight_layout()
plt.savefig('Pepcentric_DIANN_int.svg')
plt.savefig('Pepcentric_DIANN_int.png', dpi=600)
plt.show()
plt.clf()
plt.close()
